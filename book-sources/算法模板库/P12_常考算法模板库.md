## 模板 1: 基础二分查找 ⭐⭐⭐

**适用场景**：有序数组中查找目标值

```java
/**
 * 基础二分查找
 * 时间复杂度：O(log n)
 * 空间复杂度：O(1)
 *
 * 适用条件：
 * 1. 数组有序
 * 2. 可以随机访问（数组或支持索引的数据结构）
 */
public int binarySearch(int[] nums, int target) {
    if (nums == null || nums.length == 0) return -1;

    int left = 0, right = nums.length - 1;

    while (left <= right) {  // 注意：<= 不是 <
        int mid = left + (right - left) / 2;  // 防止溢出

        if (nums[mid] == target) {
            return mid;
        } else if (nums[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    return -1;  // 未找到
}
```

```java
 * 二分答案模板
 * 当问题可以转化为"找到满足某个条件的最大/最小值"时使用
 *

 * - 最小化最大值（如分配巧克力）

public int binarySearchAnswer(int low, int high) {

        if (check(mid)) {     // check(mid): 判断 mid 是否满足条件
            high = mid;       // 满足条件，尝试更小的
        } else {
            low = mid + 1;    // 不满足，需要更大的
        }
    }
    return low;  // 或 return high（此时 low == high）
}

public int minEatingSpeed(int[] piles, int h) {
    int maxPile = 0;
    for (int pile : piles) {
        maxPile = Math.max(maxPile, pile);
    }

    int left = 1, right = maxPile;
    while (left < right) {
        int mid = left + (right - left) / 2;

        if (canEatAll(piles, h, mid)) {
            right = mid;
        } else {
            left = mid + 1;
        }
    }
    return left;
}

private boolean canEatAll(int[] piles, int h, int k) {
    long hours = 0;
    for (int pile : piles) {
        hours += (pile + k - 1) / k;  // 向上取整
    }
    return hours <= h;
}
```

```java
/**
 * 左右指针模板
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
public int twoSumSorted(int[] numbers, int target) {
    int left = 0, right = numbers.length - 1;

    while (left < right) {
        int sum = numbers[left] + numbers[right];

        if (sum == target) {
            return new int[]{left + 1, right + 1};  // 题目要求 1-indexed
        } else if (sum < target) {
            left++;      // 需要更大的数
        } else {
            right--;     // 需要更小的数
        }
    }

    return new int[]{};  // 未找到
}
```

```python
    """左右指针（相向双指针）"""
    left, right = 0, len(numbers) - 1

    while left < right:
        total = numbers[left] + numbers[right]

        if total == target:

        elif total < target:
            left += 1
        else:
            right -= 1
```

```java
 * 快慢指针模板
 * 经典问题：判断链表是否有环
 */
public boolean hasCycle(ListNode head) {
    ListNode slow = head, fast = head;

    while (fast != null && fast.next != null) {
        slow = slow.next;          // 慢指针走一步
        fast = fast.next.next;     // 快指针走两步

        if (slow == fast) {
            return true;  // 相遇则有环
        }
    }

    return false;  // fast 到达尾部，无环
}
```

```java
/**
 * BFS 通用模板
 * 时间复杂度：O(V + E)
 * 空间复杂度：O(V)
 */
public int bfs(Node start, Node target) {
    Queue<Node> queue = new LinkedList<>();
    Set<Node> visited = new HashSet<>();

    queue.offer(start);
    visited.add(start);
    int steps = 0;

    while (!queue.isEmpty()) {
        int size = queue.size();  // 当前层节点数

        for (int i = 0; i < size; i++) {
            Node curr = queue.poll();

            if (curr == target) {
                return steps;  // 找到目标
            }

            // 扩展邻居
            for (Node neighbor : getNeighbors(curr)) {
                if (!visited.contains(neighbor)) {
                    visited.add(neighbor);
                    queue.offer(neighbor);
                }
            }
        }

        steps++;  // 进入下一层
    }

    return -1;  // 未找到
}
```

**适用场景**：动态连通性、环检测、最小生成树

```java
class UnionFind {
    private int[] parent;
    private int[] rank;

    public UnionFind(int n) {
        parent = new int[n];
        rank = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            rank[i] = 1;
        }
    }

    public int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);  // 路径压缩
        }
        return parent[x];
    }

    public boolean union(int x, int y) {
        int rootX = find(x), rootY = find(y);
        if (rootX == rootY) return false;

        if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
        } else if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
        } else {
            parent[rootY] = rootX;
            rank[rootX]++;
        }
        return true;
    }

    public boolean connected(int x, int y) {
        return find(x) == find(y);
    }
}
```

```python
class UnionFind:
    """并查集（路径压缩 + 按秩合并）"""

    def __init__(self, n: int):
        self.parent = list(range(n))
        self.rank = [1] * n

    def find(self, x: int) -> int:
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x: int, y: int) -> bool:
        root_x, root_y = self.find(x), self.find(y)
        if root_x == root_y:
            return False

        if self.rank[root_x] < self.rank[root_y]:
            self.parent[root_x] = root_y
        elif self.rank[root_x] > self.rank[root_y]:
            self.parent[root_y] = root_x
        else:
            self.parent[root_y] = root_x
            self.rank[root_x] += 1
        return True

    def connected(self, x: int, y: int) -> bool:
        return self.find(x) == self.find(y)
```

```java
class Trie {

    public Trie() {
        root = new TrieNode();
    }

    /** 插入单词 */
    public void insert(String word) {
        TrieNode node = root;
        for (char c : word.toCharArray()) {
            int index = c - 'a';
            if (node.children[index] == null) {
                node.children[index] = new TrieNode();
            }
            node = node.children[index];
        }
        node.isEndOfWord = true;
    }

    /** 搜索单词 */
    public boolean search(String word) {
        TrieNode node = searchPrefix(word);
        return node != null && node.isEndOfWord;
    }

    /** 判断是否有以 prefix 为前缀的单词 */
    public boolean startsWith(String prefix) {
        return searchPrefix(prefix) != null;
    }

    private TrieNode searchPrefix(String prefix) {
        TrieNode node = root;
        for (char c : prefix.toCharArray()) {
            int index = c - 'a';
            if (node.children[index] == null) {
                return null;
            }
            node = node.children[index];
        }
        return node;
    }
}
```

```java
/**
 * 拓扑排序 - Kahn 算法（基于 BFS）
 * 时间复杂度：O(V + E)
 */
public int[] topologicalSort(int numCourses, int[][] prerequisites) {
    // 构建图和入度数组
    List<List<Integer>> graph = new ArrayList<>();
    int[] inDegree = new int[numCourses];

    for (int i = 0; i < numCourses; i++) {
        graph.add(new ArrayList<>());
    }

    for (int[] pre : prerequisites) {
        graph.get(pre[1]).add(pre[0]);  // pre[1] → pre[0]
        inDegree[pre[0]]++;
    }

    // 入度为 0 的节点入队
    Queue<Integer> queue = new LinkedList<>();
    for (int i = 0; i < numCourses; i++) {
        if (inDegree[i] == 0) {
            queue.offer(i);
        }
    }

    // BFS 拓扑排序
    int[] result = new int[numCourses];
    int index = 0;

    while (!queue.isEmpty()) {
        int curr = queue.poll();
        result[index++] = curr;

        for (int next : graph.get(curr)) {
            inDegree[next]--;
            if (inDegree[next] == 0) {
                queue.offer(next);
            }
        }
    }

    // 如果存在环，返回空数组
    return index == numCourses ? result : new int[]{};
}
```
